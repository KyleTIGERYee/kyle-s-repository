3.1 对话状态管理

为什么需要状态管理

对话轮次 用户输入 提取参数 状态更新
第1轮 "我想在宝安区租房" area: 宝安区 {area: 宝安区}
第2轮 "预算2000左右" budget: 2000 {area, budget}
第3轮 "有没有一房一厅" room_type: 一房一厅 {area, budget, room_type}
第4轮 "太远了，换个福田的吧" area: 福田区 {area: 福田, budget, room_type}

在多轮对话中，需要记住用户之前提供的信息，实现参数的累积和更新

3.3 代码节点实现

状态管理代码

import json

def main(router_output, current_state_str):

# 1. 解析LLM输出

router_data = {}

try:

if isinstance(router_output, dict):

router_data = router_output

else:

raw_text = str(router_output)

start_idx = raw_text.find('{')

end_idx = raw_text.rfind('}')

if start_idx != -1 and end_idx != -1:

json_str = raw_text[start_idx:end_idx+1]

router_data = json.loads(json_str)

except Exception as e:

router_data = {"intent": "chat", "params_update": {}}

# 2. 提取数据

intent = router_data.get("intent", "chat")

new_params = router_data.get("params_update", {})

# 3. 读取旧状态

state = {}

if current_state_str:

try:

state = json.loads(str(current_state_str))

except:

state = {}

# 4. 智能合并逻辑

# 如果切换了区域(area)，则清空门店(storeNames)

if new_params.get("area") and new_params["area"] != state.get("area"):

state["storeNames"] = ""

state["address"] = ""

# 更新参数

for key, value in new_params.items():

if value is not None and str(value) != "":

state[key] = value

# 5. 准备API Payload

api_payload = state.copy()

if intent == "search_house":

if "minPrice" not in api_payload:

api_payload["minPrice"] = "0"

if "maxPrice" not in api_payload:

api_payload["maxPrice"] = "10000"

# 6. 返回结果

return {

"intent": intent,

"new_state_str": json.dumps(state, ensure_ascii=False),

"area": state.get("", ""),

"budget": state.get("", ""),

"room_type": state.get("", "")

}
